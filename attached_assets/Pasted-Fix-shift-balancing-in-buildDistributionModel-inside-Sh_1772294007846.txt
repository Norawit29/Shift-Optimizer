Fix 1: Replace maxShifts-based targets with per-shift availability-based targets
Add this new private method to the class:
tsprivate computePerShiftAvailability(shiftIdx: number): number[] {
  const D = this.daysInMonth;
  return this.staff.map((_, i) => {
    let count = 0;
    for (let d = 0; d < D; d++) {
      const req = this.getStaffPerShiftForDay(d + 1)[shiftIdx];
      if (req > 0 && !this.isBlocked(i, d, shiftIdx)) count++;
    }
    return count;
  });
}
